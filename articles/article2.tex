\section*{সেকশন: List Styles}

Using \verb|\begin{itemize} ... \end{itemize}|
\begin{itemize}
        \item Item one
        \item Item two
\end{itemize}

Using \verb|\begin{myitemize} ... \end{myitemize}|
\begin{myitemize}
        \item Item one
        \item Item two
\end{myitemize}

Using \verb|\begin{mylist} ... \end{mylist}|
\begin{mylist}
        \item Item one
        \item Item two
\end{mylist}

Using \verb|\begin{enumerate} ... \end{enumerate}|
\begin{enumerate}

        \item Item two
